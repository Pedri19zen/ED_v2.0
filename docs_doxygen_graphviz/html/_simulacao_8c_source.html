22/** @brief Conta os clientes activos que estao dentro da loja. */
23int ContarDentroLoja(Supermercado *S)
24{
25 int i, n = 0;
26 for (i = 0; i < S->clientes.total; i++)
27 if (S->clientes.v[i].ativo && S->clientes.v[i].dentroLoja) n++;
28 return n;

41/**
42 * @brief Escolhe a caixa aberta, nao cheia e com a fila mais curta.
43 *
44 * Uma caixa com LIMITE_FILA_CAIXA pessoas em espera deixa de receber
45 * novos clientes; quem acabar as compras fica em emCompras ate existir
46 * outra caixa com espaco (ou ate GerirCaixas abrir mais uma).
47 * @param excluir Caixa a ignorar (ex.: a que esta a fechar) ou NULL.
48 * @return Caixa ou NULL se nenhuma estiver disponivel.
49 */
50Caixa *EscolherMelhorCaixa(Supermercado *S, Caixa *excluir)
51{
52 Caixa *vec[MAX_CAIXAS], *melhor = NULL;
53 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
54 for (i = 0; i < n; i++) {
55 if (!vec[i]->ativa || vec[i]->aFechar || vec[i] == excluir) continue;
56 if (TamanhoFila(&vec[i]->fila) >= S->LIMITE_FILA_CAIXA) continue;
57 if (melhor == NULL || TamanhoFila(&vec[i]->fila) < TamanhoFila(&melhor->fila))
58 melhor = vec[i];
59 }
60 return melhor;
61}

90/** @brief Escolhe ao acaso um cliente registado que esteja fora da loja. */
91static Cliente *EscolherClienteParaEntrar(Supermercado *S)
92{
93 int total = S->clientes.total, inicio, i;
94 if (total == 0) return NULL;
95 inicio = Aleatorio(0, total - 1);
96 for (i = 0; i < total; i++) {
97 Cliente *c = &S->clientes.v[(inicio + i) % total];
98 if (c->ativo && !c->dentroLoja) return c;
99 }
100 return NULL;
101}

127/**
128 * @brief Passo "entrada" - tenta admitir ate ENTRADAS_POR_TICK_MAX clientes.
129 *
130 * So aceita entradas dentro do horario e enquanto S->aceitarEntradas
131 * for true. Para a loja em capacidade maxima ou sem clientes disponiveis.
132 */
133static void EntradaCliente(Supermercado *S)
134{
135 int tentativa;
136 if (!S->aceitarEntradas) return;
137 if (!LojaAberta(S)) return; /* fora de horas */
138 for (tentativa = 0; tentativa < ENTRADAS_POR_TICK_MAX; tentativa++) {
139 Cliente *c;
140 if (Aleatorio(1, 100) > S->CADENCIA_ENTRADA) continue; /* nao entrou */
141 if (ContarDentroLoja(S) >= CAPACIDADE_LOJA) return; /* loja cheia */
142 c = EscolherClienteParaEntrar(S);
143 if (c == NULL) return; /* nao ha disponiveis */
144 if (c->dentroLoja) continue; /* defensivo */
145 c->dentroLoja = true;
146 c->naFila = false;
147 c->caixaAtual[0] = '\0';
148 c->numProdutos = Aleatorio(1, MAX_CARRINHO);
149 PrepararCarrinho(c, &S->produtos, S->MAX_PRECO, S->TEMPO_ATENDIMENTO_PRODUTO);
150 EnfileirarCliente(&S->emCompras, c);
151 S->entradasDesdeUpdate++;
152 {
153 int h = (GetTempo(S->relogio) / 3600) % 24;
154 S->entradasPorHora[h]++;
155 }
156 AcrescentarEntrada(S, c);
157 if (S->verboso) printf(" [Entrou] %s (%d artigos)\n", c->nome, c->numProdutos);
158 }
159}

161/**
162 * @brief Passo "compras -> caixa".
163 *
164 * Cada cliente avanca o seu tempoComprasRestante; quando chega a zero,
165 * vai para a fila da melhor caixa (a com a fila mais curta).
166 */
167static void AvancarCompras(Supermercado *S)
168{
169 Cliente *prontos[CAPACIDADE_LOJA];
170 int nProntos = 0, i, vel = S->relogio->velocidade;
171 NoFila *p;
172 /* 1) avanca o tempo de compras de cada um; marca os que acabaram */
173 for (p = S->emCompras.inicio; p != NULL && nProntos < CAPACIDADE_LOJA; p = p->prox) {
174 p->cliente->tempoComprasRestante -= vel;
175 if (p->cliente->tempoComprasRestante <= 0)
176 prontos[nProntos++] = p->cliente;
177 }
178 /* 2) move-os para a fila da melhor caixa */
179 for (i = 0; i < nProntos; i++) {
180 Caixa *cx = EscolherMelhorCaixa(S, NULL);
181 if (cx == NULL) break; /* nenhuma caixa disponivel: fica a comprar */
182 RemoverClienteDaFila(&S->emCompras, prontos[i]);
183 EnfileirarCliente(&cx->fila, prontos[i]);
184 prontos[i]->naFila = true;
185 CopiarNome(prontos[i]->caixaAtual, cx->nome);
186 }
187}

189/**
190 * @brief Finaliza o atendimento do cliente em curso na caixa cx.
191 *
192 * Actualiza estatisticas (caixa, operador, globais), regista o cliente
193 * como atendido e poe-no a sair da loja.
194 */
195static void TerminarAtendimento(Supermercado *S, Caixa *cx)
196{
197 Cliente *c = cx->aAtender;
198 /* dinheiro que o cliente paga: o valor do carrinho menos o artigo que
199 lhe foi oferecido (se esteve a espera mais do que MAX_ESPERA). */
200 float pago = c->valorCarrinho;
201 if (c->foiOferecido) pago -= c->precoMenorProduto;
202 S->somaTemposEspera += c->tempoEspera;
203 S->totalAtendidos++;
204 S->totalProdutosVendidos += c->numProdutos;
205 S->totalDinheiro += pago;
206 cx->pessoasAtendidas++;
207 cx->produtosVendidos += c->numProdutos;
208 cx->dinheiroFeito += pago;
209 if (cx->operador != NULL) {
210 cx->operador->pessoasAtendidas++;
211 cx->operador->produtosVendidos += c->numProdutos;
212 cx->operador->dinheiroFeito += pago;
213 }
214 RegistarAtendido(cx, c->nome);
215 /* o cliente sai da loja */
216 c->dentroLoja = false;
217 c->naFila = false;
218 c->caixaAtual[0] = '\0';
219 cx->aAtender = NULL;
220 S->saidasDesdeUpdate++;
221 {
222 int h = (GetTempo(S->relogio) / 3600) % 24;
223 S->saidasPorHora[h]++;
224 }
225}

227/** @brief Fecha definitivamente uma caixa marcada com aFechar quando esvazia. */
228static void FecharCaixaSeVazia(Caixa *cx, int verboso)

238/** @brief Passo "atendimento": cada caixa aberta atende um pouco o cliente da frente. */
239static void AtenderCaixas(Supermercado *S)
240{
241 Caixa *vec[MAX_CAIXAS];
242 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
243 int vel = S->relogio->velocidade;
244 for (i = 0; i < n; i++) {
245 Caixa *cx = vec[i];
246 NoFila *p;
247 if (!cx->ativa) continue;
248 /* quem espera na fila acumula tempo de espera */
249 for (p = cx->fila.inicio; p != NULL; p = p->prox)
250 p->cliente->tempoEspera += vel;
251 /* se ninguem esta a ser atendido, chama o proximo da fila */
252 if (cx->aAtender == NULL && !FilaVazia(&cx->fila))
253 cx->aAtender = DesenfileirarCliente(&cx->fila);
254 /* atende o cliente atual */
255 if (cx->aAtender != NULL) {
256 cx->aAtender->tempoRestante -= vel;
257 if (cx->aAtender->tempoRestante <= 0)
258 TerminarAtendimento(S, cx);
259 }
260 FecharCaixaSeVazia(cx, S->verboso);
261 }
262}

264/**
265 * @brief Passo "garantia de qualidade".
266 *
267 * Oferece um produto (o mais barato do carrinho) a quem ja esperou mais
268 * do que MAX_ESPERA. Cada cliente so e' "agraciado" uma vez.
269 */
270static void VerificarTemposEspera(Supermercado *S)
271{
272 Caixa *vec[MAX_CAIXAS];
273 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
274 for (i = 0; i < n; i++) {
275 NoFila *p;
276 for (p = vec[i]->fila.inicio; p != NULL; p = p->prox) {
277 Cliente *c = p->cliente;
278 if (!c->foiOferecido && c->tempoEspera > S->MAX_ESPERA && c->numProdutos > 0) {
279 char linha[128];
280 int usado, restante;
281 c->foiOferecido = true;
282 S->produtosOferecidos++;
283 S->custoOferecido += c->precoMenorProduto;
284 S->ofertasDesdeUpdate++;
285 /* acrescenta " Nome [ esperou Ns, oferecido X.XX EUR ]" ao buffer */
286 snprintf(linha, sizeof(linha),
287 " %s [ esperou %ds, oferecido %.2f EUR ]\n",
288 c->nome, c->tempoEspera, c->precoMenorProduto);
289 usado = (int) strlen(S->nomesOfertas);
290 restante = (int) sizeof(S->nomesOfertas) - usado;
291 if ((int) strlen(linha) + 1 < restante)
292 strcat(S->nomesOfertas, linha);
293 else if (restante > 9)
294 strcat(S->nomesOfertas, " ...\n");
295 if (S->verboso)
296 printf(" [Oferta] %s esperou demais; oferta de 1 produto (%.2f EUR).\n",
297 c->nome, c->precoMenorProduto);
298 }
299 }
300 }

303/**
304 * @brief Passo "gerir caixas".
305 *
306 * Abre uma caixa se a media de clientes por fila ultrapassar MAX_FILA
307 * (requisito 5) ou marca para fechar a caixa com menos pessoas se a
308 * media for inferior a MIN_FILA (requisito 6).
309 */
310static void GerirCaixas(Supermercado *S)
311{
312 int abertas = ContarCaixasAbertas(S);
313 int emFilas = ContarClientesEmFilas(S);
314 float media;
315 if (abertas == 0) { AbrirNovaCaixa(S); return; } /* garante 1 aberta */
316 media = (float) emFilas / abertas;
317 if (media > S->MAX_FILA) {
318 AbrirNovaCaixa(S); /* req. 5: abrir automatico */
319 } else if (media < S->MIN_FILA && abertas > 1) {
320 /* req. 6: marcar para fechar a caixa com menos pessoas */
321 Caixa *vec[MAX_CAIXAS], *menos = NULL;
322 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
323 for (i = 0; i < n; i++) {
324 if (!vec[i]->ativa || vec[i]->aFechar) continue;
325 if (menos == NULL || TamanhoFila(&vec[i]->fila) < TamanhoFila(&menos->fila))
326 menos = vec[i];
327 }
328 if (menos != NULL) {
329 menos->aFechar = true;
330 if (S->verboso) printf(" [A fechar] %s (poucas pessoas)\n", menos->nome);
331 }
332 }
333 abertas = ContarCaixasAbertas(S);
334 if (abertas > S->caixasAbertasMax) S->caixasAbertasMax = abertas;
335}

357/**
358 * @brief Executa um tick completo da simulacao.
359 *
360 * Avanca o relogio, processa entradas, faz as compras, atende as caixas,
361 * verifica esperas, gere o numero de caixas e actualiza os picos.
362 */
363void ExecutarPasso(Supermercado *S)
364{
365 AvancarRelogio(S->relogio);
366 EntradaCliente(S);
367 AvancarCompras(S);
368 AtenderCaixas(S);
369 VerificarTemposEspera(S);
370 GerirCaixas(S);
371 AtualizarPicos(S);
372}

374/**
375 * @brief Corre 'nPassos' passos consecutivos da simulacao.
376 * @param nPassos Numero de passos a executar.
377 * @param comPausa Se != 0, espera 1 segundo entre passos (para observacao).
378 */
379void ExecutarSimulacao(Supermercado *S, int nPassos, int comPausa)
380{
381 int i;
382 if (ContarCaixasAbertas(S) == 0) AbrirNovaCaixa(S);
383 for (i = 0; i < nPassos; i++) {
384 ExecutarPasso(S);
385 if (comPausa) wait_segundos(1);
386 }

389/**
390 * @brief Fecha a "porta" (deixa de aceitar entradas) e atende ate esvaziar.
391 *
392 * Garante que ha pelo menos uma caixa aberta. Tem uma trava de seguranca
393 * de MAX_ITERACOES_SIM passos.
394 */
395void CorrerAteEsvaziar(Supermercado *S, int comPausa)
396{
397 int it = 0;
398 if (ContarCaixasAbertas(S) == 0) AbrirNovaCaixa(S);
399 S->aceitarEntradas = false;
400 while (!SimulacaoTerminada(S) && it < MAX_ITERACOES_SIM) {
401 ExecutarPasso(S);
402 if (comPausa) wait_segundos(1);
403 it++;
404 }
405 S->aceitarEntradas = true;
406 printf("Simulacao terminada (loja vazia) ao fim de t = %d s.\n", GetTempo(S->relogio));
407}

409/**
410 * @brief Reinicia o "dia": evacua clientes, esvazia filas, zera contadores.
411 *
412 * As caixas mantem-se com o estado actual (abertas/fechadas) e os
413 * funcionarios continuam ao servico. O relogio volta a HORA_ABERTURA.
414 */
415void IniciarNovoDia(Supermercado *S)
416{
417 Caixa *vec[MAX_CAIXAS];
418 int n, i;
419 /* clientes -> todos fora da loja */
420 for (i = 0; i < S->clientes.total; i++) {
421 Cliente *c = &S->clientes.v[i];
422 c->dentroLoja = false;
423 c->naFila = false;
424 c->caixaAtual[0] = '\0';
425 c->tempoEspera = 0;
426 c->foiOferecido = false;
427 }
428 /* caixas -> esvazia filas, lista de atendidos e zera estatisticas */
429 n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS);
430 for (i = 0; i < n; i++) {
431 Caixa *cx = vec[i];
432 NoNome *p, *seg;
433 while (!FilaVazia(&cx->fila)) DesenfileirarCliente(&cx->fila);
434 cx->aAtender = NULL;
435 cx->aFechar = false;
436 cx->pessoasAtendidas = 0;
437 cx->produtosVendidos = 0;
438 cx->dinheiroFeito = 0;
439 for (p = cx->atendidos; p != NULL; p = seg) { seg = p->prox; free(p); }
440 cx->atendidos = NULL;
441 }
442 /* lista global de quem anda as compras */
443 while (!FilaVazia(&S->emCompras)) DesenfileirarCliente(&S->emCompras);
444 /* funcionarios -> zera estatisticas (mantem ocupado/livre) */
445 for (i = 0; i < S->funcionarios.total; i++) {
446 S->funcionarios.v[i].pessoasAtendidas = 0;
447 S->funcionarios.v[i].produtosVendidos = 0;
448 S->funcionarios.v[i].dinheiroFeito = 0;
449 }
450 /* contadores globais e historico do dia */
451 S->totalAtendidos = 0;
452 S->totalProdutosVendidos = 0;
453 S->somaTemposEspera = 0;
454 S->produtosOferecidos = 0;
455 S->custoOferecido = 0;
456 S->totalDinheiro = 0;
457 S->caixasAbertasMax = 0;
458 S->maxFilaObservada = 0;
459 S->dentroLojaPico = 0;
460 S->horaPico = 0;
461 S->entradasDesdeUpdate = 0;
462 S->saidasDesdeUpdate = 0;
463 S->nomesEntradas[0] = '\0';
464 S->ofertasDesdeUpdate = 0;
465 S->nomesOfertas[0] = '\0';
466 for (i = 0; i < 24; i++) {
467 S->entradasPorHora[i] = 0;
468 S->saidasPorHora[i] = 0;
469 }
470 /* relogio na hora de abertura */
471 S->relogio->tempoAtual = S->HORA_ABERTURA * 3600;
472 printf("Novo dia iniciado as %02d:00.\n", S->HORA_ABERTURA);
473}
