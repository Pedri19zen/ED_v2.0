17/** @brief Soma dos codigos das letras modulo TAMANHO_HASH. */
18int FuncaoHash(char *nome)
19{
20 int soma = 0, i;
21 for (i = 0; nome[i] != '\0'; i++)
22 soma += (unsigned char) nome[i];
23 return soma % TAMANHO_HASH;

26/** @brief Insere uma caixa na cabeca do bucket correspondente. */
27void InserirCaixa(Hashing *h, Caixa *c)
28{
29 int pos = FuncaoHash(c->nome);
30 NoHash *novo = (NoHash *) malloc(sizeof(NoHash));
31 novo->caixa = c;
32 novo->prox = h->tabela[pos];
33 h->tabela[pos] = novo;
34 h->totalCaixas++;

37/** @brief Procura uma caixa pelo nome. */
38Caixa *PesquisarCaixa(Hashing *h, char *nome)
39{
40 int pos = FuncaoHash(nome);
41 NoHash *p = h->tabela[pos];
42 while (p != NULL) {
43 if (strcmp(p->caixa->nome, nome) == 0) return p->caixa;
44 p = p->prox;
45 }
46 return NULL;
47}

49/** @brief Copia para 'vetor' os apontadores de todas as caixas. */
50int ObterTodasCaixas(Hashing *h, Caixa *vetor[], int maxCaixas)
51{
52 int i, n = 0;
53 NoHash *p;
54 for (i = 0; i < TAMANHO_HASH; i++) {
55 p = h->tabela[i];
56 while (p != NULL && n < maxCaixas) {

64/** @brief Liberta todos os nos da tabela e as caixas que eles guardam. */
65void DestruirHashing(Hashing *h)
66{
67 int i;
68 NoHash *p, *seguinte;
69 for (i = 0; i < TAMANHO_HASH; i++) {

72 seguinte = p->prox;
73 DestruirCaixa(p->caixa);

75 p = seguinte;
76 }
77 h->tabela[i] = NULL;
