12/** @brief Cria uma caixa nova, sem operador e com a fila vazia. */
13Caixa *CriarCaixa(char *nome, bool ativa)
14{
15 Caixa *c = (Caixa *) malloc(sizeof(Caixa));
16 CopiarNome(c->nome, nome);
17 c->ativa = ativa;
18 c->aFechar = false;
19 c->operador = NULL;
20 CriarFila(&c->fila);
21 c->aAtender = NULL;
22 c->pessoasAtendidas = 0;
23 c->produtosVendidos = 0;
24 c->dinheiroFeito = 0;
25 c->atendidos = NULL;
26 return c;
27}

29/** @brief Acrescenta um nome a cabeca da lista de atendidos. */
30void RegistarAtendido(Caixa *c, char *nomeCliente)
31{
32 NoNome *novo = (NoNome *) malloc(sizeof(NoNome));
33 CopiarNome(novo->nome, nomeCliente);
34 novo->prox = c->atendidos; /* insere a cabeca da lista */
35 c->atendidos = novo;
36}

38/** @brief Imprime "Caixa -- N cliente(s) atendido(s):" + um nome por linha. */
39void ListarAtendidosCaixa(Caixa *c)
40{
41 NoNome *p;
42 int total = 0;
43 for (p = c->atendidos; p != NULL; p = p->prox) total++;
44 printf(" %s -- %d cliente(s) atendido(s):\n", c->nome, total);
45 if (c->atendidos == NULL) { printf(" (ninguem)\n"); return; }
46 for (p = c->atendidos; p != NULL; p = p->prox)
47 printf(" - %s\n", p->nome);
48}

50/** @brief Liberta a fila, a lista de atendidos e a propria caixa. */
51void DestruirCaixa(Caixa *c)
52{
53 NoNome *p = c->atendidos, *seguinte;
54 DestruirFila(&c->fila);
55 while (p != NULL) { seguinte = p->prox; free(p); p = seguinte; }
56 free(c);
57}
