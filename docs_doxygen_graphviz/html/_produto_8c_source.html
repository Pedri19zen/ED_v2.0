23/** @brief Adiciona um produto ao catalogo. @return Indice ou -1 se cheio. */
24int AdicionarProduto(ListaProdutos *L, int codigo, char *nome,
25 float preco, float tempoComprar, float tempoPagar)
26{
27 Produto *p;
28 if (L->total >= MAX_PRODUTOS) return -1;
29 p = &L->v[L->total];
30 p->codigo = codigo;
31 /* nome de produto pode ser longo: usamos snprintf com o tamanho proprio */
32 snprintf(p->nome, MAX_NOME_PRODUTO, "%s", nome);
33 p->preco = preco;
34 p->tempoComprar = tempoComprar;
35 p->tempoPagar = tempoPagar;
36 p->ativo = true;
37 L->total++;
38 InserirNomeHash(&L->idxNome, p->nome, L->total - 1);
39 return L->total - 1;
40}

42/** @brief Pesquisa por nome em O(1) media via tabela de dispersao. */
43int PesquisarProduto(ListaProdutos *L, char *nome)
44{
45 int idx = PesquisarNomeHash(&L->idxNome, nome);
46 if (idx < 0 || idx >= L->total) return -1;
47 if (!L->v[idx].ativo) return -1;
48 return idx;
49}

51/** @brief Altera o preco de um produto. */
52int EditarProduto(ListaProdutos *L, char *nome, float novoPreco)

73/**
74 * @brief Mostra o catalogo em paginas de 20 linhas.
75 *
76 * O utilizador escolhe se quer ver mais (Enter) ou sair (q).
77 */
78void ListarProdutos(ListaProdutos *L)
79{
80 int i = 0, naPagina;
81 char resp[8];
82 printf("\n--- Produtos (%d) ---\n", L->total);
83 printf(" %-6s %-50s %8s %8s %8s\n", "codigo", "nome", "preco", "comprar", "pagar");
84 while (i < L->total) {
85 naPagina = 0;
86 while (i < L->total && naPagina < 20) {
87 if (L->v[i].ativo) {
88 printf(" %-6d %-50.50s %8.2f %8.1f %8.1f\n",
89 L->v[i].codigo, L->v[i].nome,
90 L->v[i].preco, L->v[i].tempoComprar, L->v[i].tempoPagar);
91 naPagina++;
92 }
93 i++;
94 }
95 if (i < L->total) {
96 printf(" --- Enter para mais, 'q' para sair: ");
97 fflush(stdout);
98 if (fgets(resp, sizeof(resp), stdin) == NULL) break;
99 if (resp[0] == 'q' || resp[0] == 'Q') break;
100 }
101 }
102}

104/** @brief Devolve o indice de um produto activo escolhido ao acaso (ou -1). */
105int ProdutoAleatorio(ListaProdutos *L)
106{
107 int i, tentativa;
108 if (L->total == 0) return -1;
109 /* tenta algumas posicoes ao acaso; se falhar, procura o 1o ativo */
110 for (tentativa = 0; tentativa < 5; tentativa++) {
111 i = Aleatorio(0, L->total - 1);
112 if (L->v[i].ativo) return i;
113 }
114 for (i = 0; i < L->total; i++)
115 if (L->v[i].ativo) return i;
116 return -1;
117}

119/**
120 * @brief Carrega o catalogo do ficheiro.
121 *
122 * Formato (uma linha por produto, separado por TAB):
123 * codigo TAB nome TAB preco TAB tempoComprar TAB tempoPagar
124 */
125int CarregarProdutos(ListaProdutos *L, char *ficheiro)
126{
127 FILE *f = fopen(ficheiro, "r");
128 char linha[512];
129 char nome[MAX_NOME_PRODUTO];
130 int codigo;
131 float preco, tComprar, tPagar;
132 if (f == NULL) return 0;
133 while (fgets(linha, sizeof(linha), f) != NULL) {
134 /* %127[^\t] le o nome (com espacos) ate ao tabulador seguinte */
135 if (sscanf(linha, "%d\t%127[^\t]\t%f\t%f\t%f",
136 &codigo, nome, &preco, &tComprar, &tPagar) == 5)
137 AdicionarProduto(L, codigo, nome, preco, tComprar, tPagar);
138 }
139 fclose(f);
140 return 1;
141}

158/** @brief Submenu interactivo de gestao do catalogo de produtos. */
159void MenuProdutos(ListaProdutos *L)
160{
161 int op, i;
162 char nome[MAX_NOME_PRODUTO];
163 float preco, tComprar, tPagar;
164 do {
165 printf("\n=== Gerir Produtos ===\n");
166 printf("1 - Adicionar\n2 - Editar preco\n3 - Remover\n");
167 printf("4 - Pesquisar\n5 - Listar\n0 - Voltar\n");
168 op = LerOpcao("Opcao:", 0, 5);
169 switch (op) {
170 case 1:
171 LerString("Nome do produto:", nome, MAX_NOME_PRODUTO);
172 preco = LerFloat("Preco:");
173 tComprar = LerFloat("Tempo de compra (s):");
174 tPagar = LerFloat("Tempo de pagamento (s):");
175 if (nome[0] == '\0' || preco <= 0 || tComprar < 0 || tPagar < 0) {
176 printf("Dados invalidos.\n"); break;
177 }
178 /* codigo automatico: ultimo + 1, ou 90001 se vazio */
179 {
180 int codigo = (L->total > 0) ? (L->v[L->total - 1].codigo + 1) : 90001;
181 if (AdicionarProduto(L, codigo, nome, preco, tComprar, tPagar) >= 0) {
182 printf("Produto adicionado (codigo %d).\n", codigo);
183 RegistarHistorico("Adicionou produto", nome);
184 } else printf("Catalogo cheio.\n");
185 }
186 break;
187 case 2:
188 LerString("Nome do produto:", nome, MAX_NOME_PRODUTO);
189 preco = LerFloat("Novo preco:");
190 if (preco <= 0) { printf("Preco invalido.\n"); break; }
191 if (!Confirmar("Confirma a edicao deste produto?")) {
192 printf("Operacao cancelada.\n"); break;
193 }
194 if (EditarProduto(L, nome, preco)) {
195 printf("Editado.\n");
196 RegistarHistorico("Editou produto", nome);
197 } else printf("Nao encontrado.\n");
198 break;
199 case 3:
200 LerString("Nome do produto:", nome, MAX_NOME_PRODUTO);
201 if (!Confirmar("Confirma a remocao deste produto?")) {
202 printf("Operacao cancelada.\n"); break;
203 }
204 if (RemoverProduto(L, nome)) {
205 printf("Removido.\n");
206 RegistarHistorico("Removeu produto", nome);
207 } else printf("Nao encontrado.\n");
208 break;
209 case 4:
210 LerString("Nome do produto:", nome, MAX_NOME_PRODUTO);
211 i = PesquisarProduto(L, nome);
212 if (i >= 0)
213 printf("Encontrado: %d %s %.2f EUR (comprar %.1fs, pagar %.1fs)\n",
214 L->v[i].codigo, L->v[i].nome, L->v[i].preco,
215 L->v[i].tempoComprar, L->v[i].tempoPagar);
216 else printf("Nao encontrado.\n");
217 break;
218 case 5: ListarProdutos(L); break;
219 }
220 } while (op != 0);
221}
