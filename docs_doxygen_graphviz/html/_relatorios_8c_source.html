12/** @brief Converte "Caixa1" em "Caixa 1" para uma listagem mais legivel. */
13static void FormatarNomeCaixa(char *destino, char *original)
14{
15 if (strncmp(original, "Caixa", 5) == 0 && original[5] != '\0' && original[5] != ' ')
16 snprintf(destino, MAX_NOME, "Caixa %s", original + 5);
17 else
18 snprintf(destino, MAX_NOME, "%s", original);
19}

21/**
22 * @brief Imprime o "frame" da simulacao com o estado das caixas.
23 *
24 * Mostra cabecalho com hora, contadores desde a ultima actualizacao,
25 * todas as caixas planeadas (Caixa1..CaixaN), e a lista de quem entrou
26 * desde o frame anterior. Reinicia os contadores no fim.
27 */
28void VerEstadoAtual(Supermercado *S)
29{
30 int i, limiteCaixas;
31 int t, h, m, sec;
32 char nomeCaixa[MAX_NOME], nomeFmt[MAX_NOME];
33 t = GetTempo(S->relogio);
34 h = (t / 3600) % 24;
35 m = (t / 60) % 60;
36 sec = t % 60;
37 printf("\n%s[%02d:%02d:%02d] ---------- ESTADO DAS CAIXAS -------------------------%s\n",
38 COR_HDR, h, m, sec, COR_RESET);
39 printf(" Clientes na loja: %d\n", ContarDentroLoja(S));
40 printf(" Desde a ultima atualizacao: %d entraram | %d sairam\n",
41 S->entradasDesdeUpdate, S->saidasDesdeUpdate);
42 printf(" %sProdutos oferecidos ate agora: %d (%.2f EUR)%s\n",
43 COR_WARN, S->produtosOferecidos, S->custoOferecido, COR_RESET);
44 printf("%s----------------------------------------------------%s\n", COR_DIM, COR_RESET);
45 limiteCaixas = S->N_CAIXAS;
46 if (limiteCaixas > MAX_CAIXAS) limiteCaixas = MAX_CAIXAS;
47 for (i = 1; i <= limiteCaixas; i++) {
48 Caixa *cx;
49 snprintf(nomeCaixa, sizeof(nomeCaixa), "Caixa%d", i);
50 FormatarNomeCaixa(nomeFmt, nomeCaixa);
51 cx = PesquisarCaixa(&S->caixas, nomeCaixa);
52 if (cx != NULL && cx->ativa) {
53 printf(" %-8s %s[ABERTA]%s | ", nomeFmt, COR_OK, COR_RESET);
54 ImprimirBarraFila(TamanhoFila(&cx->fila));
55 printf(" | A atender: %s\n",
56 cx->aAtender ? cx->aAtender->nome : "(livre)");
57 } else {
58 printf(" %-8s %s[FECHADA]%s\n", nomeFmt, COR_DIM, COR_RESET);
59 }
60 }
61 printf("%s----------------------------------------------------%s\n", COR_DIM, COR_RESET);
62 if (S->nomesEntradas[0] != '\0') {
63 printf(" Entraram:\n%s", S->nomesEntradas);
64 }
65 if (S->nomesOfertas[0] != '\0') {
66 printf(" %sOfertas (esperaram > %ds):%s\n%s",
67 COR_WARN, S->MAX_ESPERA, COR_RESET, S->nomesOfertas);
68 }
69 /* reinicia os contadores para a proxima atualizacao */
70 S->entradasDesdeUpdate = 0;
71 S->saidasDesdeUpdate = 0;
72 S->nomesEntradas[0] = '\0';
73 S->ofertasDesdeUpdate = 0;
74 S->nomesOfertas[0] = '\0';
75}

77/** @brief Requisito 11: imprime as medidas de desempenho do sistema. */
78void MedidasDesempenho(Supermercado *S)
79{
80 Caixa *vec[MAX_CAIXAS], *maisPessoas = NULL, *maisProdutos = NULL, *maisDinheiro = NULL;
81 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
82 Funcionario *menosOp = NULL, *opMaisDinheiro = NULL;
83 int hpico_h = (S->horaPico / 3600) % 24;
84 int hpico_m = (S->horaPico / 60) % 60;
85 printf("\n===== Medidas de Desempenho =====\n");
86 for (i = 0; i < n; i++) {
87 if (maisPessoas == NULL || vec[i]->pessoasAtendidas > maisPessoas->pessoasAtendidas)
88 maisPessoas = vec[i];
89 if (maisProdutos == NULL || vec[i]->produtosVendidos > maisProdutos->produtosVendidos)
90 maisProdutos = vec[i];
91 if (maisDinheiro == NULL || vec[i]->dinheiroFeito > maisDinheiro->dinheiroFeito)
92 maisDinheiro = vec[i];
93 }
94 if (maisPessoas)
95 printf("Caixa que atendeu mais pessoas: %s (%d)\n",
96 maisPessoas->nome, maisPessoas->pessoasAtendidas);
97 if (maisProdutos)
98 printf("Caixa que vendeu mais produtos: %s (%d)\n",
99 maisProdutos->nome, maisProdutos->produtosVendidos);
100 if (maisDinheiro)
101 printf("Caixa que fez mais dinheiro: %s (%.2f EUR)\n",
102 maisDinheiro->nome, maisDinheiro->dinheiroFeito);
103 for (i = 0; i < S->funcionarios.total; i++) {
104 Funcionario *fu = &S->funcionarios.v[i];
105 if (!fu->ativo || !fu->dentroLoja) continue;
106 if (menosOp == NULL || fu->pessoasAtendidas < menosOp->pessoasAtendidas)
107 menosOp = fu;
108 if (opMaisDinheiro == NULL || fu->dinheiroFeito > opMaisDinheiro->dinheiroFeito)
109 opMaisDinheiro = fu;
110 }
111 if (menosOp)
112 printf("Operador que atendeu menos pessoas: %s (%d)\n",
113 menosOp->nome, menosOp->pessoasAtendidas);
114 if (opMaisDinheiro)
115 printf("Operador que mais dinheiro fez: %s (%.2f EUR)\n",
116 opMaisDinheiro->nome, opMaisDinheiro->dinheiroFeito);
117 printf("Produtos oferecidos (o mais barato de cada carrinho): %d | custo total: %.2f EUR\n",
118 S->produtosOferecidos, S->custoOferecido);
119 printf("Dinheiro total faturado: %.2f EUR\n", S->totalDinheiro);
120 if (S->totalAtendidos > 0)
121 printf("Tempo medio de espera: %.1f s\n",
122 (float) S->somaTemposEspera / S->totalAtendidos);
123 printf("Maior fila observada: %d pessoas\n", S->maxFilaObservada);
124 printf("Pico de clientes na loja: %d (as %02d:%02d)\n",
125 S->dentroLojaPico, hpico_h, hpico_m);
126 printf("\n-- Dinheiro por caixa --\n");
127 for (i = 0; i < n; i++)
128 printf(" %-8s: %.2f EUR (%d pessoas, %d produtos)\n",
129 vec[i]->nome, vec[i]->dinheiroFeito,
130 vec[i]->pessoasAtendidas, vec[i]->produtosVendidos);
131 printf("\n-- Pessoas atendidas por caixa --\n");
132 for (i = 0; i < n; i++)
133 ListarAtendidosCaixa(vec[i]);
134}

136/**
137 * @brief Requisito 9: bytes alocados pelo programa.
138 *
139 * Soma a struct principal, o relogio, o nome dinamico, e todos os nos
140 * alocados pelas listas/filas/tabelas de dispersao.
141 */
142long CalcularMemoriaUtilizada(Supermercado *S)
143{
144 Caixa *vec[MAX_CAIXAS];
145 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
146 long total = (long) sizeof(Supermercado) + (long) sizeof(Relogio);
147 total += (long) strlen(S->nome) + 1;
148 total += (long) S->emCompras.tamanho * (long) sizeof(NoFila);
149 for (i = 0; i < n; i++) {
150 NoNome *p;
151 total += (long) sizeof(Caixa) + (long) sizeof(NoHash);
152 total += (long) TamanhoFila(&vec[i]->fila) * (long) sizeof(NoFila);
153 for (p = vec[i]->atendidos; p != NULL; p = p->prox)
154 total += (long) sizeof(NoNome);
155 }
156 total += (long) S->clientes.idxNome.totalEntradas * (long) sizeof(NoHashNome);
157 total += (long) S->produtos.idxNome.totalEntradas * (long) sizeof(NoHashNome);
158 return total;
159}

161/**
162 * @brief Requisito 10: bytes reservados mas nao usados.
163 *
164 * Inclui os slots livres dos arrays-mestre e os buckets vazios das
165 * tabelas de dispersao.
166 */
167long CalcularMemoriaDesperdicada(Supermercado *S)
168{
169 long desp = 0;
170 int i, vazios = 0;
171 desp += (long) (MAX_PRODUTOS - S->produtos.total) * (long) sizeof(Produto);
172 desp += (long) (MAX_CLIENTES - S->clientes.total) * (long) sizeof(Cliente);
173 desp += (long) (MAX_FUNCIONARIOS - S->funcionarios.total) * (long) sizeof(Funcionario);
174 for (i = 0; i < TAMANHO_HASH; i++)
175 if (S->caixas.tabela[i] == NULL) vazios++;
176 desp += (long) vazios * (long) sizeof(NoHash *);
177 vazios = 0;
178 for (i = 0; i < TAMANHO_HASH_NOMES; i++)
179 if (S->clientes.idxNome.tabela[i] == NULL) vazios++;
180 desp += (long) vazios * (long) sizeof(NoHashNome *);
181 vazios = 0;
182 for (i = 0; i < TAMANHO_HASH_NOMES; i++)
183 if (S->produtos.idxNome.tabela[i] == NULL) vazios++;
184 desp += (long) vazios * (long) sizeof(NoHashNome *);
185 return desp;
186}

196/** @brief Requisito 2: grava um relatorio textual com o resumo da simulacao. */
197int GravarDados(Supermercado *S, char *ficheiro)
198{
199 FILE *f = fopen(ficheiro, "w");
200 Caixa *vec[MAX_CAIXAS];
201 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
202 int hpico_h, hpico_m;
203 Funcionario *opMaisDinheiro = NULL;
204 if (f == NULL) return 0;
205 hpico_h = (S->horaPico / 3600) % 24;
206 hpico_m = (S->horaPico / 60) % 60;
207 fprintf(f, "=== Relatorio da Simulacao - %s ===\n", S->nome);
208 fprintf(f, "Tempo de simulacao: %d s\n", GetTempo(S->relogio));
209 fprintf(f, "Clientes atendidos: %d\n", S->totalAtendidos);
210 fprintf(f, "Produtos vendidos: %d\n", S->totalProdutosVendidos);
211 if (S->totalAtendidos > 0)
212 fprintf(f, "Tempo medio de espera: %.1f s\n",
213 (float) S->somaTemposEspera / S->totalAtendidos);
214 fprintf(f, "Produtos oferecidos (o mais barato de cada carrinho): %d (custo %.2f EUR)\n",
215 S->produtosOferecidos, S->custoOferecido);
216 fprintf(f, "Dinheiro total faturado: %.2f EUR\n", S->totalDinheiro);
217 fprintf(f, "Maximo de caixas abertas em simultaneo: %d\n", S->caixasAbertasMax);
218 fprintf(f, "Maior fila observada: %d pessoas\n", S->maxFilaObservada);
219 fprintf(f, "Pico de clientes dentro da loja: %d (as %02d:%02d)\n",
220 S->dentroLojaPico, hpico_h, hpico_m);
221
222 for (i = 0; i < S->funcionarios.total; i++) {
223 Funcionario *fu = &S->funcionarios.v[i];
224 if (!fu->ativo) continue;
225 if (opMaisDinheiro == NULL || fu->dinheiroFeito > opMaisDinheiro->dinheiroFeito)
226 opMaisDinheiro = fu;
227 }
228 if (opMaisDinheiro)
229 fprintf(f, "Operador que mais dinheiro fez: %s (%.2f EUR)\n",
230 opMaisDinheiro->nome, opMaisDinheiro->dinheiroFeito);
231
232 fprintf(f, "\n-- Por caixa --\n");
233 for (i = 0; i < n; i++)
234 fprintf(f, "%s: atendidos=%d, produtos=%d, dinheiro=%.2f EUR, operador=%s\n",
235 vec[i]->nome, vec[i]->pessoasAtendidas, vec[i]->produtosVendidos,
236 vec[i]->dinheiroFeito,
237 vec[i]->operador ? vec[i]->operador->nome : "-");
238
239 fprintf(f, "\n-- Movimento por hora --\n");
240 fprintf(f, " hora entraram sairam\n");
241 for (i = 0; i < 24; i++)
242 if (S->entradasPorHora[i] > 0 || S->saidasPorHora[i] > 0)
243 fprintf(f, " %02d:00 %5d %5d\n", i,
244 S->entradasPorHora[i], S->saidasPorHora[i]);
245 fclose(f);
246 return 1;
247}
