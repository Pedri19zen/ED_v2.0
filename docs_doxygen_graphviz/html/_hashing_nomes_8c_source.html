17/**
18 * @brief Funcao de dispersao "djb2"-like (multiplicador 31).
19 * @param nome Cadeia a hashear.
20 * @return Indice no array de buckets.
21 */
22static int FuncaoHashNome(char *nome)
23{
24 unsigned int s = 0;
25 int i;
26 for (i = 0; nome[i] != '\0'; i++)
27 s = s * 31u + (unsigned char) nome[i];
28 return (int) (s % TAMANHO_HASH_NOMES);
29}

31/** @brief Insere o par (nome, indice) na cabeca do bucket correspondente. */
32void InserirNomeHash(HashingNomes *h, char *nome, int indice)
33{
34 int pos = FuncaoHashNome(nome);
35 NoHashNome *novo = (NoHashNome *) malloc(sizeof(NoHashNome));
36 novo->nome = nome; /* guarda o ponteiro - nao copia */
37 novo->indice = indice;
38 novo->prox = h->tabela[pos];
39 h->tabela[pos] = novo;
40 h->totalEntradas++;
41}
42
43/** @brief Devolve o indice associado ao nome, ou -1 se nao existir. */
44int PesquisarNomeHash(HashingNomes *h, char *nome)
45{
46 int pos = FuncaoHashNome(nome);
47 NoHashNome *p = h->tabela[pos];
48 while (p != NULL) {
49 if (strcmp(p->nome, nome) == 0) return p->indice;
50 p = p->prox;
51 }
52 return -1;
53}

55/** @brief Liberta todos os nos da tabela (os nomes vivem no array-mestre). */
56void DestruirHashingNomes(HashingNomes *h)
57{
58 int i;
59 NoHashNome *p, *seg;
60 for (i = 0; i < TAMANHO_HASH_NOMES; i++) {
61 p = h->tabela[i];
62 while (p != NULL) {
63 seg = p->prox;
64 free(p);
65 p = seg;
66 }
67 h->tabela[i] = NULL;
68 }
69 h->totalEntradas = 0;
70}
