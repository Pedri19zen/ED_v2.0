12/** @brief Adiciona um funcionario (por omissao ao servico e livre). */
13int AdicionarFuncionario(ListaFuncionarios *L, char *nome)
14{
15 Funcionario *f;
16 if (L->total >= MAX_FUNCIONARIOS) return -1;
17 f = &L->v[L->total];
18 f->codigo = 0; /* sera definido em CarregarFuncionarios se aplicavel */
19 CopiarNome(f->nome, nome);
20 f->ativo = true;
21 f->dentroLoja = true;
22 f->ocupado = false;
23 f->pessoasAtendidas = 0;

74/**
75 * @brief Carrega os funcionarios do ficheiro.
76 *
77 * Aceita dois formatos por linha:
78 * - "codigo TAB nome" (TSV, formato novo)
79 * - "nome" (linha simples, formato antigo)
80 */
81int CarregarFuncionarios(ListaFuncionarios *L, char *ficheiro)
82{
83 FILE *f = fopen(ficheiro, "r");
84 char linha[128], nome[MAX_NOME];
85 int codigo, idx;
86 if (f == NULL) return 0;
87 while (fgets(linha, sizeof(linha), f) != NULL) {
88 linha[strcspn(linha, "\r\n")] = '\0';
89 if (linha[0] == '\0') continue;
90 if (sscanf(linha, "%d\t%49[^\r\n]", &codigo, nome) == 2) {
91 idx = AdicionarFuncionario(L, nome);
92 if (idx >= 0) L->v[idx].codigo = codigo;
93 } else {
94 AdicionarFuncionario(L, linha); /* formato antigo: linha = nome */
95 }
96 }
97 fclose(f);
98 return 1;
99}

124/** @brief Submenu interactivo de gestao de funcionarios. */
125void MenuFuncionarios(ListaFuncionarios *L)
126{
127 int op, r;
128 char nome[MAX_NOME], novo[MAX_NOME];
129 do {
130 printf("\n=== Gerir Funcionarios ===\n");
131 printf("1 - Adicionar\n2 - Editar nome\n3 - Remover\n");
132 printf("4 - Pesquisar\n5 - Listar\n0 - Voltar\n");
133 op = LerOpcao("Opcao:", 0, 5);
134 switch (op) {
135 case 1:
136 LerString("Nome:", nome, MAX_NOME);
137 if (nome[0] == '\0') { printf("Nome invalido.\n"); break; }
138 if (PesquisarFuncionario(L, nome) >= 0) { printf("Ja existe.\n"); break; }
139 if (AdicionarFuncionario(L, nome) >= 0) {
140 printf("Funcionario adicionado.\n");
141 RegistarHistorico("Adicionou funcionario", nome);
142 } else printf("Sem espaco.\n");
143 break;
144 case 2:
145 LerString("Nome atual:", nome, MAX_NOME);
146 LerString("Novo nome:", novo, MAX_NOME);
147 if (!Confirmar("Confirma a edicao deste funcionario?")) {
148 printf("Operacao cancelada.\n"); break;
149 }
150 if (EditarFuncionario(L, nome, novo)) {
151 char det[2 * MAX_NOME + 8];
152 snprintf(det, sizeof(det), "%s -> %s", nome, novo);
153 printf("Editado.\n");
154 RegistarHistorico("Editou funcionario", det);
155 } else printf("Nao encontrado.\n");
156 break;
157 case 3:
158 LerString("Nome:", nome, MAX_NOME);
159 if (!Confirmar("Confirma a remocao deste funcionario?")) {
160 printf("Operacao cancelada.\n"); break;
161 }
162 r = RemoverFuncionario(L, nome);
163 if (r == 1) {
164 printf("Removido.\n");
165 RegistarHistorico("Removeu funcionario", nome);
166 }
167 else if (r == -1) printf("Esta a operar uma caixa.\n");
168 else printf("Nao encontrado.\n");
169 break;
170 case 4:
171 LerString("Nome:", nome, MAX_NOME);
172 printf(PesquisarFuncionario(L, nome) >= 0 ? "Encontrado.\n" : "Nao encontrado.\n");
173 break;
174 case 5: ListarFuncionarios(L); break;
175 }
176 } while (op != 0);
177}
