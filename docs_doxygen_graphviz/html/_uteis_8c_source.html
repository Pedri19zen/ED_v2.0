45/**
46 * @brief Imprime "Fila: N pessoas [●●●·······]" colorida.
47 *
48 * Verde se n <= 3, amarelo se n <= 6, vermelho se n > 6. A barra tem
49 * sempre 10 celulas.
50 */
51void ImprimirBarraFila(int n)
52{
53 int i, cheias;
54 char *cor;
55 if (n < 0) n = 0;
56 cheias = (n > 10) ? 10 : n;
57 if (n <= 3) cor = COR_OK;
58 else if (n <= 6) cor = COR_WARN;
59 else cor = COR_ERR;
60 printf("%sFila: %2d pessoas [", cor, n);
61 for (i = 0; i < cheias; i++) fputs("\xe2\x97\x8f", stdout); /* U+25CF '●' */
62 for (i = cheias; i < 10; i++) fputs("\xc2\xb7", stdout); /* U+00B7 '·' */
63 printf("]%s", COR_RESET);
64}

77int LerInteiro(char *txt)
78{
79 char linha[64];
80 int valor;
81 printf("%s ", txt);
82 fflush(stdout);
83 if (fgets(linha, sizeof(linha), stdin) == NULL) return -1;
84 if (sscanf(linha, "%d", &valor) != 1) return -1;

186/* =====================================================================
187 Deteccao de teclas em tempo real, usada pela simulacao automatica.
188 Em Windows usamos a biblioteca conio (_kbhit/_getch); noutros sistemas
189 colocamos o terminal em modo nao-canonico e lemos sem bloquear.
190 ===================================================================== */

194/** @brief Pausa 'ms' milissegundos (Sleep do WinAPI). */
195void DormirMs(int ms) { Sleep(ms); }
196
197/** @brief 1 se houver uma tecla a espera no buffer (nao bloqueia). */
198int TeclaPressionada(void) { return _kbhit(); }
199
200/** @brief Consome todas as teclas pendentes no buffer. */
201void DescartarTecla(void) { while (_kbhit()) _getch(); }
202
203#else
204
205/** @brief Pausa 'ms' milissegundos (usleep). */
206void DormirMs(int ms) { usleep(ms * 1000); }

208/** @brief Verifica, sem bloquear, se ha uma tecla a espera (consome-a se existir). */
209int TeclaPressionada(void)
210{
211 struct termios antigo, novo;
212 int ch, flags;
213 tcgetattr(STDIN_FILENO, &antigo);
214 novo = antigo;
215 novo.c_lflag &= ~(ICANON | ECHO); /* modo cru, sem eco */
216 tcsetattr(STDIN_FILENO, TCSANOW, &novo);
217 flags = fcntl(STDIN_FILENO, F_GETFL, 0);
218 fcntl(STDIN_FILENO, F_SETFL, flags | O_NONBLOCK);
219 ch = getchar();
220 tcsetattr(STDIN_FILENO, TCSANOW, &antigo); /* repoe o terminal */
221 fcntl(STDIN_FILENO, F_SETFL, flags);
222 return (ch != EOF) ? 1 : 0;
223}

230/**
231 * @brief Espera 'ms' milissegundos verificando o teclado a cada 100 ms.
232 * @return 1 se uma tecla foi premida durante a espera, 0 caso contrario.
233 */
234int EsperarOuTecla(int ms)
235{
236 int passado = 0;
237 while (passado < ms) {
238 if (TeclaPressionada()) return 1;
239 DormirMs(100);
240 passado += 100;
241 }
242 return TeclaPressionada();
243}
