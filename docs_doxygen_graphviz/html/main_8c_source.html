19/**
20 * @brief Avanca 'segundos' segundos de uma so vez, em silencio.
21 *
22 * So o estado final e' impresso (atraves de VerEstadoAtual). Usado
23 * pela opcao "saltar X segundos" do menu de pausa.
24 */
25static void SaltarSegundos(Supermercado *S, int segundos)
26{
27 int alvo = GetTempo(S->relogio) + segundos;
28 bool anterior = S->verboso;
29 S->verboso = false; /* avanco rapido e silencioso */
30 while (GetTempo(S->relogio) < alvo)
31 ExecutarPasso(S);
32 S->verboso = anterior;
33 VerEstadoAtual(S);
34}

36/**
37 * @brief Menu mostrado quando o utilizador interrompe a simulacao.
38 *
39 * Enquanto este menu estiver aberto a simulacao fica parada (o tempo
40 * nao avanca).
41 * @return 1 se a simulacao deve terminar, 0 se deve retomar.
42 */
43static int MenuPausaSimulacao(Supermercado *S)
44{
45 int op, x;
46 do {
47 printf("\n--- Simulacao PAUSADA (tempo congelado) ---\n");
48 printf("1 - Retomar simulacao\n");
49 printf("2 - Avancar 1 passo\n");
50 printf("3 - Saltar X segundos\n");
51 printf("4 - Ver estado atual\n");
52 printf("0 - Parar e voltar ao menu principal\n");
53 op = LerOpcao("Opcao:", 0, 4);
54 switch (op) {
55 case 1: return 0; /* retomar automatico */
56 case 2: ExecutarPasso(S); VerEstadoAtual(S); break;
57 case 3:
58 x = LerInteiro("Saltar quantos segundos?");
59 if (x > 0) SaltarSegundos(S, x);
60 else printf("Valor invalido.\n");
61 break;
62 case 4: VerEstadoAtual(S); break;
63 case 0: return 1; /* terminar */
64 }
65 } while (1);
66}

68/**
69 * @brief Modo "Iniciar simulacao": corre sozinha em loop.
70 *
71 * Imprime o estado a cada S->INTERVALO_AUTO_MS, ate o utilizador premir
72 * uma tecla (que abre o menu de pausa).
73 */
74static void MenuSimulacao(Supermercado *S)
75{
76 int terminar = 0, k;
77 bool verbosoAnterior = S->verboso;
78 S->verboso = false; /* so o "frame" boxed aparece durante o auto */
79 printf("\n=== Simulacao a correr ===\n");
80 printf("(Loja aberta entre %02d:00 e %02d:00. Prima uma tecla para PAUSAR.)\n",
81 S->HORA_ABERTURA, S->HORA_FECHO);
82 while (!terminar) {
83 for (k = 0; k < S->TICKS_POR_FRAME; k++) ExecutarPasso(S);
84 VerEstadoAtual(S);
85 fflush(stdout); /* mostra ja este "frame" */
86 if (!LojaAberta(S) && SimulacaoTerminada(S)) {
87 printf("\n*** Loja fechou e ja nao ha clientes. Fim do dia. ***\n");
88 break;
89 }
90 if (EsperarOuTecla(S->INTERVALO_AUTO_MS)) {
91 DescartarTecla();
92 terminar = MenuPausaSimulacao(S);
93 }
94 }
95 S->verboso = verbosoAnterior;
96 RegistarHistorico("Usou a simulacao", NULL);
97}

99/** @brief Submenu com as acoes do gerente sobre caixas e clientes. */
100static void MenuGerente(Supermercado *S)
101{
102 int op;
103 char nome[MAX_NOME], caixa[MAX_NOME], entrada[MAX_NOME];
104 do {
105 printf("\n=== Acoes do Gerente ===\n");
106 MostrarResumoCaixas(S);
107 printf("1 - Abrir nova caixa\n");
108 printf("2 - Fechar uma caixa agora (redistribui clientes)\n");
109 printf("3 - Mover cliente para outra caixa\n");
110 printf("0 - Voltar\n");
111 op = LerOpcao("Opcao:", 0, 3);
112 switch (op) {
113 case 1:
114 if (AbrirNovaCaixa(S)) RegistarHistorico("Abriu uma caixa", NULL);
115 else printf("Nao foi possivel (sem operadores livres ou limite atingido).\n");
116 break;
117 case 2:
118 LerString("Caixa a fechar (numero ou nome):", entrada, MAX_NOME);
119 ResolverNomeCaixa(entrada, caixa);
120 if (!Confirmar("Tem a certeza que quer fechar esta caixa?")) {
121 printf("Operacao cancelada.\n"); break;
122 }
123 if (FecharCaixaImediato(S, caixa)) RegistarHistorico("Fechou uma caixa", caixa);
124 break;
125 case 3:
126 LerString("Nome do cliente:", nome, MAX_NOME);
127 LerString("Caixa de destino (numero ou nome):", entrada, MAX_NOME);
128 ResolverNomeCaixa(entrada, caixa);
129 if (!Confirmar("Confirma a mudanca de caixa?")) {
130 printf("Operacao cancelada.\n"); break;
131 }
132 if (MoverClienteEntreCaixas(S, nome, caixa)) {
133 char det[2 * MAX_NOME + 8];
134 snprintf(det, sizeof(det), "%s -> %s", nome, caixa);
135 RegistarHistorico("Moveu um cliente", det);
136 }
137 break;
138 }
139 } while (op != 0);
140}
