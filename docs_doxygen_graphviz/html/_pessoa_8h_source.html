12/**
13 * @brief Um cliente do supermercado.
14 *
15 * Os campos booleanos controlam o estado durante a simulacao e evitam
16 * estados invalidos (entrar duas vezes, sair sem ter entrado, etc.).
17 */
18typedef struct {
19 int codigo; /**< Identificador (do ficheiro). */
20 char nome[MAX_NOME]; /**< Nome do cliente. */
21 int numProdutos; /**< Artigos no carrinho. */
22 float valorCarrinho; /**< Soma dos precos dos artigos. */
23 float precoMenorProduto; /**< Preco do artigo mais barato (oferecido se preciso). */
24 int tempoAtendimento; /**< Tempo total na caixa. */
25 int tempoRestante; /**< Tempo que falta enquanto atendido. */
26 int tempoEspera; /**< Tempo ja passado em fila. */
27 int tempoComprasRestante; /**< Tempo que falta para acabar as compras. */
28 bool ativo; /**< Registo existe (gestao). */
29 bool dentroLoja; /**< Esta dentro do supermercado. */
30 bool naFila; /**< Esta em fila de uma caixa. */
31 bool foiOferecido; /**< Ja recebeu um produto gratis. */
32 char caixaAtual[MAX_NOME]; /**< Nome da caixa onde esta ("" se nenhuma). */
33} Cliente;
