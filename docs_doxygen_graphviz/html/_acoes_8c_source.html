11/**
12 * @brief Requisito 5: abre uma caixa.
13 *
14 * Reutiliza uma caixa fechada se existir; se nao houver e ainda nao se
15 * atingiu N_CAIXAS, cria uma nova. Precisa de um operador livre.
16 * Os clientes ja em fila nas outras caixas NAO sao movidos -- a nova
17 * caixa comeca vazia e recebe naturalmente os clientes que acabam as
18 * compras (via EscolherMelhorCaixa) e ainda nao chegaram ao LIMITE_FILA_CAIXA
19 * das antigas.
20 * @return 1 se conseguiu, 0 caso contrario.
21 */
22int AbrirNovaCaixa(Supermercado *S)
23{
24 Caixa *vec[MAX_CAIXAS], *fechada = NULL;
25 int n = ObterTodasCaixas(&S->caixas, vec, MAX_CAIXAS), i;
26 Funcionario *op;
27 char nome[MAX_NOME];
28 for (i = 0; i < n; i++)
29 if (!vec[i]->ativa) { fechada = vec[i]; break; }
30 op = ObterFuncionarioLivre(&S->funcionarios);
31 if (op == NULL) return 0; /* sem operadores disponiveis */
32 if (fechada != NULL) { /* reabre uma caixa fechada */
33 fechada->ativa = true;
34 fechada->aFechar = false;
35 fechada->operador = op; op->ocupado = true;
36 if (S->verboso) printf(" [Caixa aberta] %s\n", fechada->nome);
37 return 1;
38 }
39 if (S->caixas.totalCaixas < S->N_CAIXAS) { /* cria uma caixa nova */
40 Caixa *nova;
41 snprintf(nome, MAX_NOME, "Caixa%d", S->caixas.totalCaixas + 1);
42 nova = CriarCaixa(nome, true);
43 nova->operador = op; op->ocupado = true;
44 InserirCaixa(&S->caixas, nova);
45 if (S->verboso) printf(" [Caixa aberta] %s (nova)\n", nome);
46 return 1;
47 }
48 return 0;
49}

51/**
52 * @brief Escolhe uma caixa de destino com espaco; se necessario tenta abrir uma.
53 *
54 * Usado apenas no fecho imediato: se todas as restantes caixas estiverem
55 * cheias, tenta criar/reabrir outra antes de desistir.
56 */
57static Caixa *EscolherDestinoOuAbrir(Supermercado *S, Caixa *excluir)
58{
59 Caixa *destino = EscolherMelhorCaixa(S, excluir);
60 if (destino == NULL && AbrirNovaCaixa(S))
61 destino = EscolherMelhorCaixa(S, excluir);
62 return destino;
63}

65/**
66 * @brief Requisito 7: fecha uma caixa de imediato.
67 *
68 * Distribui os clientes em fila pelas restantes caixas abertas, respeitando
69 * LIMITE_FILA_CAIXA. Se nao houver destino com espaco, a caixa nao fecha.
70 */
71int FecharCaixaImediato(Supermercado *S, char *nomeCaixa)
72{
73 Caixa *cx = PesquisarCaixa(&S->caixas, nomeCaixa);
74 Caixa *destino;
75 Cliente *c;
76 if (cx == NULL) { printf("Caixa nao encontrada.\n"); return 0; }
77 if (!cx->ativa) { printf("Essa caixa ja esta fechada.\n"); return 0; }
78 if (ContarCaixasAbertas(S) <= 1) {
79 /* nao deixar a loja sem caixas: tenta abrir outra primeiro */
80 if (!AbrirNovaCaixa(S)) {
81 printf("Nao ha outra caixa para receber os clientes.\n");
82 return 0;
83 }
84 }
85 /* move o cliente que estava a ser atendido */
86 if (cx->aAtender != NULL) {
87 destino = EscolherDestinoOuAbrir(S, cx);
88 if (destino != NULL) {
89 c = cx->aAtender;
90 cx->aAtender = NULL;
91 EnfileirarCliente(&destino->fila, c);
92 CopiarNome(c->caixaAtual, destino->nome);
93 } else {
94 printf("Nao ha caixa com espaco para receber os clientes.\n");
95 return 0;
96 }
97 }
98 /* move os que estavam em espera */
99 while (!FilaVazia(&cx->fila)) {
100 c = DesenfileirarCliente(&cx->fila);
101 destino = EscolherDestinoOuAbrir(S, cx);
102 if (destino == NULL) {
103 EnfileirarCliente(&cx->fila, c);
104 printf("Nao ha caixa com espaco para receber todos os clientes.\n");
105 return 0;
106 }
107 EnfileirarCliente(&destino->fila, c);
108 CopiarNome(c->caixaAtual, destino->nome);
109 }
110 cx->ativa = false;
111 cx->aFechar = false;
112 if (cx->operador != NULL) { cx->operador->ocupado = false; cx->operador = NULL; }
113 printf("Caixa %s fechada e clientes redistribuidos.\n", cx->nome);
114 return 1;
115}

117/** @brief Requisito 4: muda um cliente que esta numa fila para outra caixa. */
118int MoverClienteEntreCaixas(Supermercado *S, char *nomeCliente, char *nomeCaixa)
119{
120 int idx = PesquisarCliente(&S->clientes, nomeCliente);
121 Cliente *c;
122 Caixa *destino, *origem;
123 if (idx < 0) { printf("Cliente nao encontrado.\n"); return 0; }
124 c = &S->clientes.v[idx];
125 if (!c->dentroLoja || !c->naFila) { printf("O cliente nao esta numa fila.\n"); return 0; }
126 destino = PesquisarCaixa(&S->caixas, nomeCaixa);
127 if (destino == NULL || !destino->ativa) { printf("Caixa de destino invalida.\n"); return 0; }
128 if (strcmp(c->caixaAtual, nomeCaixa) == 0) { printf("Ja esta nessa caixa.\n"); return 0; }
129 if (TamanhoFila(&destino->fila) >= S->LIMITE_FILA_CAIXA) {
130 printf("Caixa de destino cheia.\n");
131 return 0;
132 }
133 origem = PesquisarCaixa(&S->caixas, c->caixaAtual);
134 if (origem != NULL) {
135 if (origem->aAtender == c) origem->aAtender = NULL;
136 else RemoverClienteDaFila(&origem->fila, c);
137 }
138 EnfileirarCliente(&destino->fila, c);
139 CopiarNome(c->caixaAtual, nomeCaixa);
140 printf("Cliente %s movido para %s.\n", nomeCliente, nomeCaixa);
141 return 1;
142}

144/**
145 * @brief Requisito 8: diz onde esta uma pessoa.
146 *
147 * Resultado possivel: fora, a fazer compras, em fila numa caixa, ou a
148 * ser atendida. Aceita variacoes em maiusculas/minusculas e
149 * correspondencia por prefixo.
150 */
151void PesquisarPessoa(Supermercado *S, char *nomeCliente)
152{
153 int idx = PesquisarClienteTolerante(&S->clientes, nomeCliente);
154 Cliente *c;
155 Caixa *cx;
156 if (idx < 0) { printf("Cliente nao registado.\n"); return; }
157 c = &S->clientes.v[idx];
158 if (strcmp(c->nome, nomeCliente) != 0)
159 printf("(correspondencia: '%s')\n", c->nome);
160 if (!c->dentroLoja) { printf("%s esta fora da loja.\n", c->nome); return; }
161 if (!c->naFila) { printf("%s esta a fazer compras.\n", c->nome); return; }
162 cx = PesquisarCaixa(&S->caixas, c->caixaAtual);
163 if (cx != NULL && cx->aAtender == c)
164 printf("%s esta a ser atendido na %s.\n", c->nome, c->caixaAtual);
165 else
166 printf("%s esta em espera na %s.\n", c->nome, c->caixaAtual);
167}

177/**
178 * @brief Resumo multi-linha de todas as caixas.
179 *
180 * Mostra Caixa1..CaixaN, mesmo que algumas ainda nao tenham sido criadas.
181 * As caixas inexistentes aparecem como fechadas.
182 */
183void MostrarResumoCaixas(Supermercado *S)
184{
185 int i, limiteCaixas = S->N_CAIXAS;
186 char nomeCaixa[MAX_NOME];
187 if (limiteCaixas > MAX_CAIXAS) limiteCaixas = MAX_CAIXAS;
188 printf(" %sCaixas disponiveis:%s\n", COR_HDR, COR_RESET);
189 for (i = 1; i <= limiteCaixas; i++) {
190 Caixa *cx;
191 snprintf(nomeCaixa, sizeof(nomeCaixa), "Caixa%d", i);
192 cx = PesquisarCaixa(&S->caixas, nomeCaixa);
193 if (cx != NULL && cx->ativa) {
194 printf(" %-8s %s[ABERTA]%s ", nomeCaixa, COR_OK, COR_RESET);
195 ImprimirBarraFila(TamanhoFila(&cx->fila));
196 printf("\n");
197 } else {
198 printf(" %-8s %s[FECHADA]%s\n", nomeCaixa, COR_DIM, COR_RESET);
199 }
200 }
201}

204 * @brief Aceita "Caixa1" ou simplesmente "1" como referencia a uma caixa.
205 * @param entrada Texto introduzido pelo utilizador.
206 * @param destino Buffer onde se escreve o nome canonico.

208void ResolverNomeCaixa(char *entrada, char *destino)
209{
210 char *fim;
211 long n;
212 n = strtol(entrada, &fim, 10);
213 if (entrada[0] != '\0' && *fim == '\0' && n > 0 && n < MAX_CAIXAS)
214 snprintf(destino, MAX_NOME, "Caixa%ld", n);
215 else
216 snprintf(destino, MAX_NOME, "%s", entrada);
217}
