23/**
24 * @brief Adiciona um cliente registado (ainda fora da loja).
25 * @return Indice no array ou -1 se cheio.
26 */
27int AdicionarCliente(ListaClientes *L, char *nome, int numProdutos)
28{
29 Cliente *c;
30 if (L->total >= MAX_CLIENTES) return -1;
31 c = &L->v[L->total];
32 c->codigo = 0; /* sera definido por CarregarClientes, se houver */
33 CopiarNome(c->nome, nome);
34 c->numProdutos = numProdutos;
35 c->valorCarrinho = 0;
36 c->precoMenorProduto = 0;
37 c->tempoAtendimento = 0;
38 c->tempoRestante = 0;
39 c->tempoEspera = 0;
40 c->tempoComprasRestante = 0;
41 c->ativo = true;
42 c->dentroLoja = false;
43 c->naFila = false;
44 c->foiOferecido = false;
45 c->caixaAtual[0] = '\0';
46 L->total++;
47 /* indexa o nome (aponta para o buffer no array-mestre) */
48 InserirNomeHash(&L->idxNome, c->nome, L->total - 1);
49 return L->total - 1;
50}

52/**
53 * @brief Pesquisa por nome em O(1) media usando a tabela de dispersao.
54 *
55 * So devolve indice se o cliente ainda estiver "activo".
56 */
57int PesquisarCliente(ListaClientes *L, char *nome)
58{
59 int idx = PesquisarNomeHash(&L->idxNome, nome);
60 if (idx < 0 || idx >= L->total) return -1;
61 if (!L->v[idx].ativo) return -1;
62 return idx;

65/** @brief Verifica se 'prefixo' e' prefixo de 'str' ignorando maiusculas. */
66static int PrefixoIgualSemCaso(char *str, char *prefixo)
67{
68 int i;
69 for (i = 0; prefixo[i] != '\0'; i++) {
70 if (str[i] == '\0') return 0;
71 if (ToMaiscula(str[i]) != ToMaiscula(prefixo[i])) return 0;
72 }
73 return 1;
74}

76/**
77 * @brief Pesquisa tolerante: match exacto via hash, ou prefixo sem caso.
78 *
79 * Primeiro tenta match exacto (rapido, via hash); se nao encontrar, faz
80 * um varrimento linear case-insensitive com prefixo. Devolve o primeiro
81 * cliente activo que comece pelas mesmas letras (qualquer caixa).
82 */
83int PesquisarClienteTolerante(ListaClientes *L, char *texto)
84{
85 int idx = PesquisarCliente(L, texto);
86 int i;
87 if (idx >= 0) return idx;
88 if (texto[0] == '\0') return -1;
89 for (i = 0; i < L->total; i++) {
90 if (!L->v[i].ativo) continue;
91 if (PrefixoIgualSemCaso(L->v[i].nome, texto))
92 return i;
93 }
94 return -1;
95}

119/**
120 * @brief Lista os clientes registados em paginas de 20.
121 *
122 * O utilizador pode pedir a pagina seguinte (Enter) ou sair (q).

124void ListarClientes(ListaClientes *L)
125{
126 int i = 0, naPagina;
127 char resp[8];
128 printf("\n--- Clientes registados (%d) ---\n", L->total);
129 while (i < L->total) {
130 naPagina = 0;
131 while (i < L->total && naPagina < 20) {
132 if (L->v[i].ativo) {
133 printf(" %-30s | artigos: %2d | %s\n",
134 L->v[i].nome, L->v[i].numProdutos,
135 L->v[i].dentroLoja ? (L->v[i].naFila ? "em fila" : "as compras") : "fora");
136 naPagina++;
137 }
138 i++;
139 }
140 if (i < L->total) {
141 printf(" --- Enter para mais, 'q' para sair: ");
142 fflush(stdout);
143 if (fgets(resp, sizeof(resp), stdin) == NULL) break;
144 if (resp[0] == 'q' || resp[0] == 'Q') break;
145 }
146 }
147}

149/**
150 * @brief Carrega a "pool" de clientes do ficheiro.
151 *
152 * Formato (TAB-separado): "codigo TAB nome". As terminacoes \\r\\n
153 * (Windows) sao tratadas.
154 * @return 1 se conseguiu abrir, 0 caso contrario.
155 */
156int CarregarClientes(ListaClientes *L, char *ficheiro)
157{
158 FILE *f = fopen(ficheiro, "r");
159 char linha[128], nome[MAX_NOME];
160 int codigo, idx;
161 if (f == NULL) return 0;
162 while (fgets(linha, sizeof(linha), f) != NULL) {
163 /* %[^\t\r\n] le o nome ate ao tab/CR/LF */
164 if (sscanf(linha, "%d\t%49[^\t\r\n]", &codigo, nome) == 2) {
165 idx = AdicionarCliente(L, nome, 0); /* numProdutos definido na entrada */
166 if (idx >= 0) L->v[idx].codigo = codigo;
167 }
168 }
169 fclose(f);
170 return 1;
171}

184/**
185 * @brief Prepara o carrinho do cliente.
186 *
187 * Para cada artigo (numProdutos) escolhe um produto do catalogo e soma
188 * o preco; o tempo de passagem de cada produto e' um valor em
189 * [2, TEMPO_ATENDIMENTO_PRODUTO]. Se 'prods' for NULL usa valores
190 * aleatorios como fallback.
191 */
192void PrepararCarrinho(Cliente *c, ListaProdutos *prods, int maxPreco, int tempoAtendProduto)
193{
194 int k, idx;
195 float preco, somaPagar = 0.0f, somaComprar = 0.0f;
196 if (c->numProdutos < 1) c->numProdutos = 1;
197 c->valorCarrinho = 0;
198 c->precoMenorProduto = -1; /* sentinela: ainda nao ha artigos */
199 for (k = 0; k < c->numProdutos; k++) {
200 idx = (prods != NULL) ? ProdutoAleatorio(prods) : -1;
201 if (idx >= 0) {
202 /* usa os tempos especificos do produto */
203 preco = prods->v[idx].preco;
204 somaPagar += prods->v[idx].tempoPagar;
205 somaComprar += prods->v[idx].tempoComprar;
206 } else {
207 /* sem catalogo: usa valores aleatorios como fallback */
208 preco = (float) Aleatorio(1, maxPreco);
209 somaPagar += (float) Aleatorio(2, tempoAtendProduto);
210 somaComprar += (float) Aleatorio(2, tempoAtendProduto);
211 }
212 c->valorCarrinho += preco;
213 /* guarda o preco do artigo mais barato (sera o oferecido se preciso) */
214 if (c->precoMenorProduto < 0 || preco < c->precoMenorProduto)
215 c->precoMenorProduto = preco;
216 }
217 if (c->precoMenorProduto < 0) c->precoMenorProduto = 0;
218 /* arredonda os tempos para segundos inteiros; garante pelo menos 1 segundo */
219 c->tempoAtendimento = (int)(somaPagar + 0.5f);
220 c->tempoComprasRestante = (int)(somaComprar + 0.5f);
221 if (c->tempoAtendimento < 1) c->tempoAtendimento = 1;
222 if (c->tempoComprasRestante < 1) c->tempoComprasRestante = 1;
223 c->tempoRestante = c->tempoAtendimento;
224 c->tempoEspera = 0;
225 c->foiOferecido = false;
226}

228/** @brief Submenu interactivo de gestao de clientes. */
229void MenuClientes(ListaClientes *L)
230{
231 int op, n, i, r;
232 char nome[MAX_NOME];
233 do {
234 printf("\n=== Gerir Clientes ===\n");
235 printf("1 - Adicionar\n2 - Editar nr. de produtos\n3 - Remover\n");
236 printf("4 - Pesquisar\n5 - Listar\n6 - Gerar aleatorios\n0 - Voltar\n");
237 op = LerOpcao("Opcao:", 0, 6);
238 switch (op) {
239 case 1:
240 LerString("Nome:", nome, MAX_NOME);
241 n = LerInteiro("Nr. de produtos:");
242 if (nome[0] == '\0' || n < 1) { printf("Dados invalidos.\n"); break; }
243 if (PesquisarCliente(L, nome) >= 0) { printf("Ja existe.\n"); break; }
244 if (AdicionarCliente(L, nome, n) >= 0) {
245 printf("Cliente adicionado.\n");
246 RegistarHistorico("Adicionou cliente", nome);
247 } else printf("Sem espaco.\n");
248 break;
249 case 2:
250 LerString("Nome:", nome, MAX_NOME);
251 n = LerInteiro("Novo nr. de produtos:");
252 if (n < 1) { printf("Valor invalido.\n"); break; }
253 if (!Confirmar("Confirma a edicao deste cliente?")) {
254 printf("Operacao cancelada.\n"); break;
255 }
256 if (EditarCliente(L, nome, n)) {
257 printf("Editado.\n");
258 RegistarHistorico("Editou cliente", nome);
259 } else printf("Nao encontrado.\n");
260 break;
261 case 3:
262 LerString("Nome:", nome, MAX_NOME);
263 if (!Confirmar("Confirma a remocao deste cliente?")) {
264 printf("Operacao cancelada.\n"); break;
265 }
266 r = RemoverCliente(L, nome);
267 if (r == 1) {
268 printf("Removido.\n");
269 RegistarHistorico("Removeu cliente", nome);
270 }
271 else if (r == -1) printf("Esta na loja, nao pode ser removido agora.\n");
272 else printf("Nao encontrado.\n");
273 break;
274 case 4:
275 LerString("Nome:", nome, MAX_NOME);
276 i = PesquisarCliente(L, nome);
277 if (i >= 0) printf("Encontrado: %s (%d artigos)\n", L->v[i].nome, L->v[i].numProdutos);
278 else printf("Nao encontrado.\n");
279 break;
280 case 5: ListarClientes(L); break;
281 case 6:
282 n = LerInteiro("Quantos clientes gerar?");
283 if (n > 0) {
284 char det[32];
285 GerarClientesAleatorios(L, n, 10);
286 snprintf(det, sizeof(det), "%d clientes", n);
287 printf("Gerados.\n");
288 RegistarHistorico("Gerou clientes aleatorios", det);
289 } else printf("Valor invalido.\n");
290 break;
291 }
292 } while (op != 0);
293}
