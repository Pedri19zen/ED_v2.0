6 * Os modulos sao:

24/**
25 * @brief Estrutura central do projecto.
26 *
27 * Junta os arrays-mestre carregados uma vez dos ficheiros, as estruturas
28 * dinamicas (fila de quem anda as compras e tabela de dispersao das caixas),
29 * o relogio logico, os parametros de configuracao e as estatisticas globais.
30 */
31typedef struct {
32 char *nome; /**< Nome do supermercado. */
33
34 ListaProdutos produtos; /**< Catalogo + indice de pesquisa. */
35 ListaClientes clientes; /**< Pool + indice de pesquisa. */
36 ListaFuncionarios funcionarios; /**< Operadores de caixa. */
37
38 Fila emCompras; /**< Clientes a fazer compras (sem caixa ainda). */
39 Hashing caixas; /**< Todas as caixas, acessiveis pelo nome. */
40 Relogio *relogio; /**< Relogio logico da simulacao. */
41
42 /* ---- configuracao (lida de Configuracao.txt) ---- */
43 int MAX_ESPERA; /**< Limiar para oferecer um produto. */
44 int N_CAIXAS; /**< Numero maximo de caixas. */
45 int TEMPO_ATENDIMENTO_PRODUTO; /**< Limite [2, este] de tempo por produto. */
46 int MAX_PRECO; /**< Preco maximo de um produto. */
47 int MAX_FILA; /**< Media de fila para abrir nova caixa. */
48 int MIN_FILA; /**< Media de fila para fechar uma caixa. */
49 int LIMITE_FILA_CAIXA; /**< Cap rigido de pessoas por caixa (default 10). */
50 int VELOCIDADE_RELOGIO; /**< Segundos por tick. */
51 int HORA_ABERTURA; /**< 24h: hora a que a loja abre. */
52 int HORA_FECHO; /**< 24h: hora a que a loja fecha. */
53 int CADENCIA_ENTRADA; /**< Probabilidade (%) de entrada por tentativa. */
54 int INTERVALO_AUTO_MS; /**< Periodo (ms) entre frames do auto-display. */
55 int TICKS_POR_FRAME; /**< Passos avancados entre frames. */
56
57 bool aceitarEntradas; /**< false = porta fechada (atende ate esvaziar). */
58 bool verboso; /**< true = imprime eventos passo a passo. */
59
60 /* ---- estatisticas globais ---- */
61 int totalAtendidos; /**< Total de clientes atendidos. */
62 int totalProdutosVendidos; /**< Total de produtos vendidos. */
63 long somaTemposEspera; /**< Soma dos tempos de espera. */
64 int produtosOferecidos; /**< Numero de ofertas (req. qualidade). */
65 float custoOferecido; /**< Custo total das ofertas. */
66 float totalDinheiro; /**< Faturacao total. */
67 int caixasAbertasMax; /**< Maximo de caixas abertas em simultaneo. */
68
69 /* ---- contadores reiniciados a cada VerEstadoAtual ---- */
70 int entradasDesdeUpdate; /**< Entradas desde o ultimo frame. */
71 int saidasDesdeUpdate; /**< Saidas desde o ultimo frame. */
72 char nomesEntradas[1024]; /**< Nomes acumulados (req. 2: quem entrou). */
73 int ofertasDesdeUpdate; /**< Ofertas (MAX_ESPERA) desde o ultimo frame. */
74 char nomesOfertas[1024]; /**< Linhas "Nome [ -X.XX EUR ]" acumuladas. */
75
76 /* ---- metricas adicionais ---- */
77 int maxFilaObservada; /**< Maior fila vista em qualquer caixa. */
78 int dentroLojaPico; /**< Pico de clientes dentro da loja. */
79 int horaPico; /**< Segundos absolutos do pico. */
80 int entradasPorHora[24]; /**< Histograma de entradas. */
81 int saidasPorHora[24]; /**< Histograma de saidas. */
82} Supermercado;
