16/**
17 * @brief Garante que cada produto carregado respeita os limites da config.
18 *
19 * Preco em ]0, MAX_PRECO] e tempos em [2, TEMPO_ATENDIMENTO_PRODUTO].
20 */
21static void ClamparProdutos(Supermercado *S)
22{
23 int i;
24 float maxPreco = (float) S->MAX_PRECO;
25 float maxTempo = (float) S->TEMPO_ATENDIMENTO_PRODUTO;
26 for (i = 0; i < S->produtos.total; i++) {
27 Produto *p = &S->produtos.v[i];
28 if (p->preco <= 0) p->preco = 1;
29 if (p->preco > maxPreco) p->preco = maxPreco;
30 if (p->tempoComprar < 2) p->tempoComprar = 2;
31 if (p->tempoComprar > maxTempo) p->tempoComprar = maxTempo;
32 if (p->tempoPagar < 2) p->tempoPagar = 2;
33 if (p->tempoPagar > maxTempo) p->tempoPagar = maxTempo;
34 }
35}

59/**
60 * @brief Aloca e inicializa um Supermercado com valores por omissao.
61 * @param nome Nome do supermercado (copiado para memoria propria).
62 * @return Apontador para a nova instancia.
63 */
64Supermercado *CriarSupermercado(char *nome)
65{
66 Supermercado *S = (Supermercado *) malloc(sizeof(Supermercado));
67 S->nome = (char *) malloc(strlen(nome) + 1);
68 strcpy(S->nome, nome);
69 CriarListaProdutos(&S->produtos);
70 CriarListaClientes(&S->clientes);
71 CriarListaFuncionarios(&S->funcionarios);
72 CriarFila(&S->emCompras);
73 CriarHashing(&S->caixas);
74 /* valores por omissao (caso falte o ficheiro de configuracao) */
75 S->MAX_ESPERA = 120; S->N_CAIXAS = 6; S->TEMPO_ATENDIMENTO_PRODUTO = 6;
76 S->MAX_PRECO = 40; S->MAX_FILA = 7; S->MIN_FILA = 3;
77 S->LIMITE_FILA_CAIXA = 10;
78 S->VELOCIDADE_RELOGIO = 10;
79 S->HORA_ABERTURA = 8; S->HORA_FECHO = 20;
80 S->CADENCIA_ENTRADA = 70;
81 S->INTERVALO_AUTO_MS = 2000;
82 S->TICKS_POR_FRAME = 3;
83 S->relogio = CriarRelogio(S->VELOCIDADE_RELOGIO);
84 S->aceitarEntradas = true;
85 S->verboso = true;
86 S->totalAtendidos = 0; S->totalProdutosVendidos = 0; S->somaTemposEspera = 0;
87 S->produtosOferecidos = 0; S->custoOferecido = 0; S->caixasAbertasMax = 0;
88 S->totalDinheiro = 0;
89 S->entradasDesdeUpdate = 0;
90 S->saidasDesdeUpdate = 0;
91 S->nomesEntradas[0] = '\0';
92 S->ofertasDesdeUpdate = 0;
93 S->nomesOfertas[0] = '\0';
94 S->maxFilaObservada = 0;
95 S->dentroLojaPico = 0;
96 S->horaPico = 0;
97 {
98 int k;
99 for (k = 0; k < 24; k++) {
100 S->entradasPorHora[k] = 0;
101 S->saidasPorHora[k] = 0;
102 }
103 }
104 return S;

107/**
108 * @brief Le o ficheiro de configuracao.
109 *
110 * Pares "CHAVE valor", um por linha. Linhas iniciadas por '#' sao
111 * tratadas como comentarios e ignoradas.
112 * @return 1 se conseguiu abrir o ficheiro, 0 caso contrario.
113 */
114int CarregarConfiguracao(Supermercado *S, char *ficheiro)
115{
116 FILE *f = fopen(ficheiro, "r");
117 char linha[128], chave[64];
118 int valor;
119 if (f == NULL) return 0;
120 while (fgets(linha, sizeof(linha), f) != NULL) {
121 if (linha[0] == '#' || linha[0] == '\n') continue;
122 if (sscanf(linha, "%63s %d", chave, &valor) == 2)
123 AplicarConfig(S, chave, valor);
124 }
125 fclose(f);
126 return 1;
127}

129/**
130 * @brief Cria a Caixa1 (unica caixa inicial, activa, com primeiro operador livre).
131 *
132 * O sistema arranca sempre com apenas uma caixa aberta; as restantes
133 * (ate N_CAIXAS) sao abertas dinamicamente por GerirCaixas / o gerente
134 * quando a media de fila ultrapassar MAX_FILA.
135 */
136static void CriarCaixaInicial(Supermercado *S)
137{
138 Caixa *cx = CriarCaixa("Caixa1", true);
139 Funcionario *op = ObterFuncionarioLivre(&S->funcionarios);
140 if (op != NULL) { cx->operador = op; op->ocupado = true; }
141 InserirCaixa(&S->caixas, cx);
142}

144/** @brief Le todos os ficheiros, prepara o relogio e cria a Caixa1 inicial. */
145int InicializarSupermercado(Supermercado *S)
146{
147 CarregarConfiguracao(S, FICH_CONFIG);
148 if (S->LIMITE_FILA_CAIXA < 1) S->LIMITE_FILA_CAIXA = 10;
149 /* sincroniza a velocidade do relogio com o valor (talvez) lido da config */
150 S->relogio->velocidade = S->VELOCIDADE_RELOGIO;
151 CarregarProdutos(&S->produtos, FICH_PRODUTOS);
152 ClamparProdutos(S); /* impoe limites da config */
153 CarregarFuncionarios(&S->funcionarios, FICH_FUNCIONARIOS);
154 CarregarClientes(&S->clientes, FICH_CLIENTES);
155 CriarCaixaInicial(S); /* arranca so com Caixa1 */
156 if (S->clientes.total < POOL_CLIENTES_MIN)
157 GerarClientesAleatorios(&S->clientes, POOL_CLIENTES_MIN - S->clientes.total, MAX_CARRINHO);
158 /* poe o relogio a hora de abertura (ex.: 08:00) */
159 S->relogio->tempoAtual = S->HORA_ABERTURA * 3600;
160 return 1;
161}

163/**
164 * @brief Liberta toda a memoria dinamica.
165 *
166 * Os arrays-mestre ficam dentro da struct principal, por isso sao
167 * libertados com ela.
168 */
169void DestruirSupermercado(Supermercado *S)
170{
171 DestruirHashing(&S->caixas); /* caixas, filas e listas de atendidos */
172 DestruirFila(&S->emCompras); /* nos da lista de compras */
173 DestruirListaClientes(&S->clientes); /* tabela de indices de nomes */
174 DestruirListaProdutos(&S->produtos); /* tabela de indices de nomes */
175 DestruirRelogio(S->relogio);
176 free(S->nome);
177 free(S);
178}
