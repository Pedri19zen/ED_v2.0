18/** @brief Acrescenta um cliente ao fim da fila. */
19void EnfileirarCliente(Fila *f, Cliente *c)
20{
21 NoFila *novo = (NoFila *) malloc(sizeof(NoFila));
22 novo->cliente = c;
23 novo->prox = NULL;
24 if (f->fim == NULL) f->inicio = f->fim = novo; /* fila estava vazia */
25 else { f->fim->prox = novo; f->fim = novo; }
26 f->tamanho++;

35/** @brief Retira e devolve o cliente do inicio da fila. */
36Cliente *DesenfileirarCliente(Fila *f)
37{
38 NoFila *no;
39 Cliente *c;
40 if (f->inicio == NULL) return NULL;
41 no = f->inicio;
42 c = no->cliente;
43 f->inicio = no->prox;
44 if (f->inicio == NULL) f->fim = NULL;
45 free(no);
46 f->tamanho--;
47 return c;
48}

50/**
51 * @brief Remove um cliente especifico da fila.
52 * @return 1 se encontrou e removeu, 0 caso contrario.
53 */
54int RemoverClienteDaFila(Fila *f, Cliente *c)
55{
56 NoFila *atual = f->inicio, *anterior = NULL;
57 while (atual != NULL) {
58 if (atual->cliente == c) {
59 if (anterior == NULL) f->inicio = atual->prox;
60 else anterior->prox = atual->prox;
61 if (atual == f->fim) f->fim = anterior;
62 free(atual);
63 f->tamanho--;
64 return 1;
65 }
66 anterior = atual;
67 atual = atual->prox;
68 }
69 return 0;
70}

81 f->inicio = f->fim = NULL;
82 f->tamanho = 0;
83}
